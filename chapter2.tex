\section{Иллюстрации.}

Рассмотрим следующие примеры решения обыкновенных дифференциальных уравнений. Мы нашли их точное решение либо приближённое решение с помощью метода Рунге-Кутта четвёртого порядка и решение методом Лагранжа. Данные вычислений занесли в таблицы.

\underline{Пример 1.}
\begin{center}$y'=e^t+3t+y;~y(0)=0$ \end{center}

1 способ. Найдём точное решение уравнения.

\begin{equation*}
\begin{aligned}
&y'-y=e^t+3t\\
1)~ & y'-y=0\\
&\dfrac{dy}{dt}=y\\
&\ln|y|=t+C\\
&y=Ce^t\\
2)~ & y(t)=C(t)e^t\\
&y'(t)=C'(t)e^t+C(t)e^t\\
\end{aligned}
\end{equation*}

Последние два выражения подставим в исходное уравнение:
\begin{equation*}
\begin{aligned}
&C'(t)e^t+C(t)e^t=e^t+3t+C(t)e^t\\
&C'(t)=1+3te^{-t}\\
&C(t)=\int\left(1+3te^{-t}\right)dt=t-3\int tde^{-t}=t-3\left(te^{-t}-\int e^{-t}dt\right)\\
&C(t)=t-3te^{-t}-3e^{-t}+C_1
y=(t-3te^{-t}-3e^{-t}+C)e^t
\end{aligned}
\end{equation*}

Получаем $y=te^t-3t-3+Ce^t.$

Подставляя начальное условие, получим следующий ответ:
$$
y(t)=(t+3)e^t-3t-3.
$$

2 способ. Введём новую переменную по формуле $y_1=\dfrac{t^{k_1}}{y}$ или $y=\dfrac{t^{k_1}}{y_1}$.

Находим $y_{10}$:
$$
y_{10}=\lim\limits_{t\to 0}\dfrac{t^{k_1}}{y}=\lim\limits_{t\to 0}\dfrac{k_1 t^{k_1-1}}{y'}.
$$

При $k_1=1~y_{10}=1.$

Отсюда получаем, что $y=\dfrac{t}{y_1}.$

Из последнего выражения для $y$ находим
\begin{equation}
\label{eq:pr11}
y'=\dfrac{y_1-ty_1'}{y_1^2}.
\end{equation}

Подставим выражение \eqref{eq:pr11} в исходное уравнение:
\begin{equation*}
\begin{aligned}
&\dfrac{y_1-ty_1'}{y_1^2}=e^t+3t+\dfrac{t}{y_1}\\
&y_1-ty_1'=e^ty_1^2+3ty_1^2+ty_1\\
&y_1'=\dfrac{y_1}{t}-\dfrac{e^ty_1^2}{t}-3y_1^2-y_1
\end{aligned}
\end{equation*}

Таким образом получаем новое уравнение:
\begin{equation}
\label{eq:neweq2}
y_1'=\dfrac{1}{t}(y_1-e^ty_1^2)-3y_1^2-y_1,~y_{10}=1.
\end{equation}

Опять введём новую переменную $y_2=\dfrac{t^{k_2}}{y_1-1}$ или 
\begin{equation}
\label{eq:newper}
y_1=\dfrac{t^{k_2}}{y_2}+1
\end{equation}

Находим $y_{20}$:
$$
y_{20}=\lim\limits_{t\to 0}\dfrac{t^{k_2}}{y_1-1}=\lim\limits_{t\to 0}\dfrac{k_2 t^{k_2-1}}{y_1'}.
$$

Вычислим отдельно $\lim\limits_{t\to 0}y_1'$:
\begin{equation*}
\begin{aligned}
&\lim\limits_{t\to 0}y_1'=\lim\limits_{t\to 0}\dfrac{1}{t}(y_1-e^ty_1^2)-3y_1^2-y_1=-4+\lim\limits_{t\to 0}(y_1'-e^ty_1^2-2e^ty_1y_1')\\
&\lim\limits_{t\to 0}y_1'=-5-\lim\limits_{t\to 0}y_1'\\
& 2\lim\limits_{t\to 0}y_1'=-5\\
& \lim\limits_{t\to 0}y_1'=-\dfrac{5}{2}
\end{aligned}
\end{equation*}

При $k_2=1~y_{20}=-\dfrac{2}{5}.$ Отсюда получаем, что $y_1=\dfrac{t}{y_2}+1.$

Из выражения \eqref{eq:newper} $y_1'=\dfrac{y_2-ty_2'}{y_2^2}.$ Подставляем в уравнение \eqref{eq:neweq2}:
\begin{equation*}
\begin{aligned}
&\dfrac{y_2-ty_2'}{y_2^2}=\dfrac{1}{t}\left(\dfrac{t}{y_2}+1-e^t\left(\dfrac{t}{y_2}+1\right)^2\right)-3\left(\dfrac{t}{y_2}+1\right)^2-\dfrac{t}{y_2}-1\\
&\dfrac{y_2-ty_2'}{y_2^2}=\dfrac{1}{y_2}+\dfrac{1}{t}-\dfrac{te^t}{y_2^2}-\dfrac{2e^t}{y_2}-\dfrac{e^t}{t}-\dfrac{3t^2}{y_2^2}-\dfrac{7t}{y_2}-4\\
& y_2-ty_2'=y_2+\dfrac{y_2^2}{t}-te^t-2e^ty_2-\dfrac{e^ty_2^2}{t}-3t^2-7ty_2-4y_2^2\\
& y_2'=-\dfrac{y_2^2}{t^2}+e^t+2\dfrac{e^ty_2}{t}+\dfrac{e^ty_2^2}{t^2}+3t+7y_2+\dfrac{4y_2^2}{t}
\end{aligned}
\end{equation*}

Таким образом, имеем дифференциальное уравнение:
$$
y_2'=e^t+3t+7y_2+\dfrac{1}{t}\left(2e^ty_2+\dfrac{4}{5}y_2^2\right)+\dfrac{1}{t^2}(-y_2^2+e^ty_2^2),~y_{20}=-\dfrac{2}{5}
$$
Опять введём новую переменную $y_3=\dfrac{t^{k_3}}{y_2+\dfrac{2}{5}}.$

Находим $y_{30}:$
$$
y_{30}=\lim\limits_{t\to 0}\dfrac{t^{k_3}}{y_2+\dfrac{2}{5}}=\lim\limits_{t\to 0}\dfrac{k_3 t^{k_3-1}}{y_2'}.
$$

Рассмотрим отдельно $\lim\limits_{t\to 0}y_2'$:
\begin{equation*}
\begin{aligned}
&\lim\limits_{t\to 0}y_2'=1-\dfrac{14}{5}+\lim\limits_{t\to 0}\dfrac{2e^ty_2+7y_2^2}{t}+\lim\limits_{t\to 0}\dfrac{-y_2^2+e^ty_2^2}{t^2}\\
&\lim\limits_{t\to 0}y_2'=-\dfrac{11}{5}-\dfrac{2}{5}\lim\limits_{t\to 0}(2e^t+7y_2')+\dfrac{4}{25}\lim\limits_{t\to 0}\dfrac{e^t}{2t}\\
& 5\lim\limits_{t\to 0}y_2'=-\dfrac{21}{5}\\
& \lim\limits_{t\to 0}y_2'=-\dfrac{21}{25}
\end{aligned}
\end{equation*}

При $k_3=1~y_{30}=-\dfrac{25}{21}$.
 
Тогда решение данного уравнения представляется в виде следующей цепной дроби 
\begin{equation*}
y(t)= \dfrac{t}{1} \downplus \dfrac{t}{-\frac{2}{5}} \downplus \dfrac{t}{-\frac{25}{21}}\downplus \downdots
\end{equation*}

Откуда $y_{прибл}=\dfrac{21t^2+10t}{10-4t}$.

\underline{Пример 2.}
\begin{center}$y'=(2t+1)y'=4t+2y;~y(0)=2$ \end{center}

1 способ. Найдём точное решение уравнения.

\begin{equation*}
\begin{aligned}
&2ty'+y'-2y=4t\\
1)~ & 2tdy+dy-2ydt=0\\
&(2t+1)dy-2ydt=0\\
&\dfrac{dy}{y}-\dfrac{2dt}{2t+1}=0\\
&\ln |y|-2 \int \dfrac{dt}{2t+1}=C\\
& \ln \left| \dfrac{y}{2t+1}\right|=C\\
& y = C(2t+1)\\
\end{aligned}
\end{equation*}
\begin{equation*}
\begin{aligned}
2)~ & y(t)=C(t)(2t+1)\\
&y'(t)=C'(t)(2t+1)+2C(t)\\
\end{aligned}
\end{equation*}

Последние два выражения подставим в исходное уравнение:
\begin{equation*}
\begin{aligned}
&2tC'(t)(2t+1)+4tC(t)+C'(t)(2t+1)+2C(t)-2C(t)(2t+1)-4t=0\\
&C'(t)=\dfrac{4t}{(2t+1)^2}\\
&C(t)=\int \dfrac{4tdt}{(2t+1)^2}dt=\ln |2t+1|+\dfrac{1}{2t+1}+C_1\\
\end{aligned}
\end{equation*}

Получаем $y(t)=(\ln |2t+1|+C)(2t+1)+1.$

Подставляя начальное условие, получим следующий ответ:
$$
y(t)=(\ln |2t+1|+1)(2t+1)+1.
$$

2 способ. Введём новую переменную по формуле $y_1=\dfrac{t^{k_1}}{y-2}$~~~ или $y=\dfrac{t^{k_1}}{y_1}+2$.

Находим $y_{10}$:
$$
y_{10}=\lim\limits_{t\to 0}\dfrac{t^{k_1}}{y-2}=\lim\limits_{t\to 0}\dfrac{k_1 t^{k_1-1}}{y'}.
$$

При $k_1=1~y_{10}=\dfrac{1}{4}.$

Отсюда получаем, что $y=\dfrac{t}{y_1}+2.$

Из последнего выражения для $y$ находим
\begin{equation}
\label{eq:pr21}
y'=\dfrac{y_1-ty_1'}{y_1^2}.
\end{equation}

Подставим это выражение в исходное уравнение:
\begin{equation*}
\begin{aligned}
&(2t+1)\dfrac{y_1-ty_1'}{y_1^2}=4t+2\left(2+\dfrac{t}{y_1}\right)\\
&\dfrac{y_1-ty_1'}{y_1^2}=\dfrac{4t+4}{2t+1}+\dfrac{2t}{(2t+1)y_1}\\
&ty_1'=y_1-\dfrac{4ty_1^2}{2t+1}-\dfrac{4y_1^2}{2t+1}-\dfrac{2ty_1}{2t+1}
\end{aligned}
\end{equation*}

Таким образом получаем новое уравнение:
\begin{equation}
\label{eq:neweq3}
y_1'=\dfrac{y_1}{t}-\dfrac{4y_1^2}{2t+1}-\dfrac{2y_1}{2t+1}=\dfrac{y_1-4y_1^2}{t(2t+1)},~y_{10}=\dfrac{1}{4}.
\end{equation}

Опять введём новую переменную $y_2=\dfrac{t^{k_2}}{y_1-\dfrac{1}{4}}$ или 
\begin{equation}
\label{eq:newper2}
y_1=\dfrac{t^{k_2}}{y_2}+\dfrac{1}{4}
\end{equation}

Находим $y_{20}$:
$$
y_{20}=\lim\limits_{t\to 0}\dfrac{t^{k_2}}{y_1-\dfrac{1}{4}}=\lim\limits_{t\to 0}\dfrac{k_2 t^{k_2-1}}{y_1'}.
$$

Вычислим отдельно $\lim\limits_{t\to 0}y_1'(t)$:
\begin{equation*}
\begin{aligned}
&\lim\limits_{t\to 0}y_1'=\lim\limits_{t\to 0}\dfrac{y_1-4y_1^2}{2t^2+t}-\dfrac{1}{4}=-\dfrac{1}{4}+\dfrac{1}{4}\lim\limits_{t\to 0}\dfrac{1-4y_1}{2t^2+t}\\
&\lim\limits_{t\to 0}y_1'=-\dfrac{1}{4}-\lim\limits_{t\to 0}y_1'\\
& 2\lim\limits_{t\to 0}y_1'=-\dfrac{1}{4}\\
& \lim\limits_{t\to 0}y_1'=-\dfrac{1}{8}
\end{aligned}
\end{equation*}

При $k_2=1~y_{20}=-8.$ Отсюда получаем, что $y_1=\dfrac{t}{y_2}+\dfrac{1}{4}.$

Находим производную функции $y_1$: $y_1'=\dfrac{y_2-ty_2'}{y_2^2}.$ Подставляем в уравнение \eqref{eq:neweq3}:
\begin{equation*}
\begin{aligned}
&\dfrac{y_2-ty_2'}{y_2^2}=\dfrac{1}{4t}+\dfrac{1}{y_2}-\dfrac{3}{4(2t+1)}-\dfrac{4t}{y_2(2t+1)}-\\
&-\dfrac{4t^2}{y_2^2(2t+1)}-\dfrac{1}{4t(2t+1)}-\dfrac{2}{y_2(2t+1)}-\dfrac{4t}{y_2^2(2t+1)}\\
\end{aligned}
\end{equation*}

Таким образом, имеем дифференциальное уравнение:
$$
y_2'=\dfrac{y_2^2}{4t(2t+1)}+\dfrac{4y_2}{2t+1}+\dfrac{4t}{2t+1}+\dfrac{2y_2}{t(2t+1)}+\dfrac{4}{2t+1},~y_{20}=-8.
$$
Опять введём новую переменную $y_3=\dfrac{t^{k_3}}{y_2+8}.$

Находим $y_{30}:$
$$
y_{30}=\lim\limits_{t\to 0}\dfrac{t^{k_3}}{y_2+8}=\lim\limits_{t\to 0}\dfrac{k_3 t^{k_3-1}}{y_2'}.
$$

Рассмотрим отдельно $\lim\limits_{t\to 0}y_2'$:
\begin{equation*}
\begin{aligned}
&\lim\limits_{t\to 0}y_2'=4\lim\limits_{t\to 0}\dfrac{y_2}{t+1}-2\lim\limits_{t\to 0}\dfrac{y_2+8}{t(2t+1)}\\
&\lim\limits_{t\to 0}y_2'=-32-2\lim\limits_{t\to 0}y_2'\\
& \lim\limits_{t\to 0}y_2'=-\dfrac{32}{3}
\end{aligned}
\end{equation*}

При $k_3=1~y_{30}=-\dfrac{3}{32}$.
 
Тогда решение данного уравнения представляется в виде следующей цепной дроби 
\begin{equation*}
y(t)= 2+\dfrac{t}{\frac{1}{4}} \downplus \dfrac{t}{-8} \downplus \dfrac{t}{-\frac{3}{32}}\downplus \downdots
\end{equation*}

Откуда $y_{прибл}=\dfrac{32t^2+34t+12}{5t+6}$.

\underline{Пример 3.} 
\begin{center}$y'=\sin y+\cos y;~y(0)=0$ \end{center}

Решение. Введём новую переменную по формуле $y_1=\dfrac{t^{k_1}}{y}$~~~ или $y=\dfrac{t^{k_1}}{y_1}$.

Находим $y_{10}$:
$$
y_{10}=\lim\limits_{t\to 0}\dfrac{t^{k_1}}{y}=\lim\limits_{t\to 0}\dfrac{k_1 t^{k_1-1}}{y'}.
$$

При $k_1=1~y_{10}=1.$

Отсюда получаем, что $y=\dfrac{t}{y_1}.$

Из последнего выражения для $y$ находим
\begin{equation}
\label{eq:pr31}
y'=\dfrac{y_1-ty_1'}{y_1^2}.
\end{equation}

Подставим это выражение в исходное уравнение:
\begin{equation*}
\begin{aligned}
&\dfrac{y_1-ty_1'}{y_1^2}=\sin \dfrac{t}{y_1}+\cos \dfrac{t}{y_1}\\
&y_1-ty_1'=y_1^2\left(\sin \dfrac{t}{y_1}+\cos\dfrac{t}{y_1}\right)\\
\end{aligned}
\end{equation*}

Таким образом получаем новое уравнение:
\begin{equation}
\label{eq:neweq4}
y_1'=\dfrac{y_1}{t}-\dfrac{y_1^2}{t}\left(\sin \dfrac{t}{y_1}+\cos\dfrac{t}{y_1}\right),~y_{10}=1.
\end{equation}

Опять введём новую переменную $y_2=\dfrac{t^{k_2}}{y_1-1}$ или 
\begin{equation}
\label{eq:newper2}
y_1=\dfrac{t^{k_2}}{y_2}+1
\end{equation}

Находим $y_{20}$:
$$
y_{20}=\lim\limits_{t\to 0}\dfrac{t^{k_2}}{y_1-1}=\lim\limits_{t\to 0}\dfrac{k_2 t^{k_2-1}}{y_1'}.
$$

Вычислим отдельно $\lim\limits_{t\to 0}y_1'(t)$:
\begin{equation*}
\begin{aligned}
&\lim\limits_{t\to 0}y_1'=\lim\limits_{t\to 0}\dfrac{y_1-y_1^2\left(\sin \dfrac{t}{y_1}+\cos\dfrac{t}{y_1}\right)}{t}\\
&\lim\limits_{t\to 0}y_1'=\lim\limits_{t\to 0}\dfrac{y_1(1-y_1)}{t}=\lim\limits_{t\to 0}(y_1'(1-y_1)-y_1y_1')\\
& 2\lim\limits_{t\to 0}y_1'=-1\\
& \lim\limits_{t\to 0}y_1'=-\dfrac{1}{2}
\end{aligned}
\end{equation*}

При $k_2=1~y_{20}=-2.$ Отсюда получаем, что $y_1=\dfrac{t}{y_2}+1.$

Находим производную функции $y_1$: $y_1'=\dfrac{y_2-ty_2'}{y_2^2}.$ Подставляем в уравнение \eqref{eq:neweq4}:
\begin{equation*}
\begin{aligned}
&\dfrac{y_2-ty_2'}{y_2^2}=\dfrac{1}{t}\left( \dfrac{t}{y_2}+1-\left(\dfrac{t}{y_2}\right)^2\left( \sin\dfrac{t}{\dfrac{t}{y_2}+1}+\cos\dfrac{t}{\dfrac{t}{y_2}+1}\right)\right)\\
&y_2-ty_2'=\dfrac{y_2^2}{t}\left(\dfrac{t}{y_2}+1+\left(\dfrac{t^2}{y_2^2}+\dfrac{2t}{y_2}+1\right)\left(\sin\dfrac{ty_2}{t+y_2}+\cos \dfrac{ty_2}{t+y_2}\right)\right)\\
\end{aligned}
\end{equation*}

Таким образом, имеем дифференциальное уравнение:
$$
y_2'=-\dfrac{1}{t}-\left(\dfrac{t}{y_2^2}+\dfrac{2}{y_2}+\dfrac{1}{t}\right)\left(\sin\dfrac{ty_2}{t+y_2}+\cos \dfrac{ty_2}{t+y_2}\right),~y_{20}=-2.
$$
Опять введём новую переменную $y_3=\dfrac{t^{k_3}}{y_2+2}.$

Находим $y_{30}:$
$$
y_{30}=\lim\limits_{t\to 0}\dfrac{t^{k_3}}{y_2+2}=\lim\limits_{t\to 0}\dfrac{k_3 t^{k_3-1}}{y_2'}.
$$

Рассмотрим отдельно $\lim\limits_{t\to 0}y_2'$:
\begin{equation*}
\begin{aligned}
&\lim\limits_{t\to 0}y_2'=\lim\limits_{t\to 0}\dfrac{-1-\left(\dfrac{t^2}{y_2^2}+\dfrac{2t}{y_2}+1\right)\left(\sin\dfrac{ty_2}{t+y_2}+\cos \dfrac{ty_2}{t+y_2}\right)}{t}\\
&\lim\limits_{t\to 0}y_2'=\lim\limits_{t\to 0}\dfrac{-1-\sin\dfrac{ty_2}{t+y_2}-1}{t}\\
&\lim\limits_{t\to 0}y_2'=-2-\lim\limits_{t\to 0}y_2'\\
& \lim\limits_{t\to 0}y_2'=-1
\end{aligned}
\end{equation*}

При $k_3=1~y_{30}=-1$.
Тогда решение данного уравнения представляется в виде следующей цепной дроби 
\begin{equation*}
y(t)= \dfrac{t}{1} \downplus \dfrac{t}{-2} \downplus \dfrac{t}{-1}\downplus \downdots
\end{equation*}

Откуда $y_{прибл}=\dfrac{2t+t^2}{2}$.
