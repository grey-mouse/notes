\section{Иллюстрации.}
Рассмотрим следующие примеры решения систем двух дифференциальных уравнений. Мы нашли их точное решение либо приближённое решение с помощью метода Рунге-Кутта четвёртого порядка и решение методом Лагранжа. Данные вычислений занесли в таблицы.

\underline{Пример 1.} 
\begin{equation}
\label{eq:h1}
\left\{
\begin{aligned}
x'=4x-3y+\sin x,~x(0)=x_{0}=1,\\
y'=2x-y-2\cos t,~y(0)=y_{0}=2
\end{aligned}
\right.
\end{equation}

1 способ. Второе уравнение умножаем на (-3) и складываем с первым, получим
\begin{equation*}
\left\{
\begin{aligned}
x'-3y'=-2x+6 \cos t+\sin t,\\
y'=2x-y-2\cos t
\end{aligned}
\right.
\end{equation*}
\begin{equation}
\label{eq:h2}
\left\{
\begin{aligned}
3y'=x'+2x-6 \cos t-\sin t,\\
y'=2x-y-2\cos t
\end{aligned}
\right.
\end{equation}
Из уравнения \eqref{eq:h1} $3y=-x'+4x+\sin t$ или $3y'=-x''+4x'+\cos t$.
Приравнивая правые части уравнений получим:
$$
x'+2x-6\cos t-\sin t=-x''+4x'+\cos t
$$
$$ x''-3x'+2x=7\cos t+\sin t, $$
$$1)~x''-3x'+2x=0,$$
$$\lambda^2-3\lambda+2=0, $$
$$\lambda_1=2, \lambda_2=1. $$
$$ x_{o.o}=c_1e^t+c_2e^{2t}$$
$$2)~ x_{ч.н}=A\cos t+ B\sin t,$$
$$ x'=-A\sin t+B \cos t,$$
$$ x''=-A \cos t-B\sin t.$$
Подставляя эти значения в уравнение, получим
$$ -A\cos t-B\sin t+3A\sin t-3B\cos t+2A\cos t+2B\sin t=7\cos t+\sin t,$$
\begin{equation*}
\left\{
\begin{aligned}
-A-3B+2A=7,\\
-B+3A+2B=1.
\end{aligned}
\right.
\end{equation*}
\begin{equation*}
\left\{
\begin{aligned}
A=1,\\
B=-2.
\end{aligned}
\right.
\end{equation*}
$$x_{о.н}=c_1e^t+c_2e^{2t}+\cos t-2\sin t, $$
$$ y_{о.н}=\dfrac{1}{3}(-x'+4x+\sin t)$$
$$y_{о.н}=\dfrac{1}{3}(-c_1e^t-2c_2e^{2t}+\sin t+2\cos t+4c_1e^t+4\cos t-8\sin t+\sin t),$$
$$ y_{о.н}=c_1e^t+\dfrac{2}{3}c_2e^{2t}+2\cos t-2\sin t.
$$
Подставляя начальные условия, получим следующий ответ
\begin{equation*}
\left\{
\begin{aligned}
x=\cos t-2\sin t,\\
y=2\cos t-2 \sin t.
\end{aligned}
\right.
\end{equation*}

2 способ.

Введём новые переменные по формулам
$$y_1=\dfrac{t}{x-1},~x_1=\dfrac{y-2}{x-1}, $$
или
$$x=1+\dfrac{t}{y_1},~y=2+\dfrac{tx_1}{y_1}. $$
Находим $y_{10}$ и $x_{10}$:
$$ y_{10}=\lim\limits_{t\to 0}\dfrac{t}{x-1}=\lim\limits_{t\to 0}\dfrac{1}{x'}=\dfrac{1}{4-6}=-\dfrac{1}{2}.$$
$$ x_{10}=\lim\limits_{t\to 0}\dfrac{y-2}{x-1}=\lim\limits_{t\to 0}\dfrac{y'}{x'}=\lim\limits_{t\to 0}\dfrac{-2\cos t}{-2+\sin t}=1.$$
Из последних выражений для $y$ и $x$ находим
$$ x'=\dfrac{y_1-ty_1'}{y_1^2},$$
$$ y'=\dfrac{(x_1+tx_1')y_1-tx_1y_1'}{y_1^2}.$$
Подставим выражения для производных в исходное уравнение:
$$ \dfrac{y_1-ty_1'}{y_1^2}=4+\dfrac{4t}{y_1}-6-\dfrac{3tx_1}{y_1}+\sin t,$$
$$ y_1-ty_1'=-2y_1^2+4ty_1-3tx_1y_1+y_1^2\sin t,$$
$$ y_1'=\dfrac{1}{t}(y_1+2y_1^2-4ty_1+3tx_1y_1-y_1^2\sin t).$$
Аналогично
$$\dfrac{(x_1+tx_1')y_1-tx_1y_1'}{y_1^2}=2+\dfrac{2t}{y_1}-2-\dfrac{tx_1}{y_1}-2\cos t, $$
$$(x_1+tx_1')y_1-tx_1y_1'=2ty_1-tx_1y_1-2y_1^2\cos t,$$
$$(x_1+tx_1')y_1=x_1y_1+2x_1y_1^2-4txy+3tx_1^2y_1-x_1y_1^2\sin t+2ty_1-tx_1y_1\cos t,$$
$$x_1+tx_1'=x_1+2x_1y_1-5tx_1+3tx_1^2-x_1y_1\sin t+2t-2y_1\cos t,$$
$$x_1'=\dfrac{1}{t}(2x_1y_1-5tx_1+3tx_1^2-x_1y_1\sin t+2t-2y_1\cos t)$$
Таким образом, имеем систему
\begin{equation*}
\left\{
\begin{aligned}
x_1'=\dfrac{1}{t}(2x_1y_1-x_1y_1\sin t-2y_1\cos t)-5x_1+3x_1^2,~x_{10}=x_1(0)=1,\\

\end{aligned}
\right.
\end{equation*}
Опять введём новые переменные
$$y_2=\dfrac{t}{x_1-x_{10}}=\dfrac{t}{x_1-1},~~x_2=\dfrac{y_1+\dfrac{1}{2}}{x_1-1},$$
или
$$x_1=1+\dfrac{t}{y_2},~y_1=-\dfrac{1}{2}+\dfrac{tx_2}{y_2}$$
Находим $y_{20}$ и $x_{20}$:
$$y_{20}=\lim\limits_{t\to 0}\dfrac{t}{x_1-1}=\lim\limits_{t\to 0}\dfrac{1}{x_1'}$$
Вычислим отдельно $\lim\limits_{t\to 0}x_1':$
$$\lim\limits_{t\to 0}x_1'=\lim\limits_{t\to 0}\dfrac{2x_1y_1-x_1y_1\sin t-2y_1\cos t}{t}-5+3+2+\lim\limits_{t\to 0}\dfrac{2y_1(x_1-\cos t)}{t}$$
$$\lim\limits_{t\to 0}x_1'=\dfrac{1}{2}-\lim\limits_{t\to 0}\dfrac{x_1'+\sin t}{1}=\dfrac{1}{2}-\lim\limits_{t\to 0}x_1'.$$
$$2\lim\limits_{t\to 0}x_1'=\dfrac{1}{2}$$
$$\lim\limits_{t\to 0}x_1'=\dfrac{1}{4}$$
Тогда $y_{20}=4$.
$$x_{20}=\lim\limits_{t\to 0}\dfrac{y_1'}{x_1'}=4\lim\limits_{t\to 0}y_1'=4(2-\dfrac{3}{2}+\lim\limits_{t\to 0}\dfrac{y_1(1+2y_1)}{t}-\dfrac{1}{4})=4(\dfrac{1}{4}-\dfrac{1}{2}\lim\limits_{t\to 0}2y_1')$$
Тогда $\lim\limits_{t\to 0}y_1'=\dfrac{1}{4}-\lim\limits_{t\to 0}y_1'$, то есть $\lim\limits_{t\to 0}y_1'=\dfrac{1}{8}$. Получаем, что $x_{20}=4\cdot \dfrac{1}{8}=\dfrac{1}{2}.$

\underline{Пример 2.} 
\begin{equation}
\label{eq:h2}
\left\{
\begin{aligned}
x'=2x+y+2t,~x(0)=x_{0}=1,\\
y'=x+2y+t,~y(0)=y_{0}=1
\end{aligned}
\right.
\end{equation}

1 способ. Найдём точное решение.

Второе уравнение умножаем на (-2) и складываем спервым, получим
\begin{equation*}
\left\{
\begin{aligned}
x'=2y'-3y,\\
y'=x+2y+t
\end{aligned}
\right.
\end{equation*}
Из первого уравнения этой системы получим $x=y'-2y-t$ или $$x'=y''-2y'-1$$.
Приравнивая правые части второго уравнения системы и последнего уравнения, получим
$$2y'-3y=y''-2y'-1$$
$$y''-4y'+3y=1.$$
$$1)~y''-4y'+3y=0$$
$$\lambda^2-4\lambda+3=0$$
$$\lambda_1=3, \lambda_2=1.$$
$$y_{o.o}=c_1e^{3t}+c_2e^t$$
$$2)~y_{ч.н}=A,$$
$$A=\dfrac{1}{3}$$
$$y_{о.н}=c_1e^{3t}+c_2e^t+\dfrac{1}{3}$$
$$x_{о.н}=y'-2y-t=3c_1e^{3t}+c_2e^t-t-2c_1e^{3t}-2c_2e^t-\dfrac{2}{3}=c_1e^{3t}-c_e^t-t-\dfrac{2}{3}.$$
Подставляя начальные условия, получим
$$
\left\{
\begin{aligned}
c_1-c_2-\dfrac{2}{3}=1,\\
c_1+c_2+\dfrac{1}{3}=1.
\end{aligned}
\right.
$$
$$
\left\{
\begin{aligned}
c_1=\dfrac{7}{6},\\
c_2=-\dfrac{1}{2}.
\end{aligned}
\right.
$$
То есть
$$
\left\{
\begin{aligned}
x=\dfrac{7}{6}e^{3t}+\dfrac{1}{2}e^t-t-\dfrac{2}{3},\\
y=\dfrac{7}{6}e^{3t}-\dfrac{1}{2}e^t+\dfrac{1}{3}.
\end{aligned}
\right.
$$

2 способ.Введём новые переменные по формулам
$$y_1=\dfrac{t}{x-1},~x_1=\dfrac{y-1}{x-1}, $$
или
$$x=1+\dfrac{t}{y_1},~y=1+\dfrac{tx_1}{y_1}. $$
Находим $y_{10}$ и $x_{10}$:
$$ y_{10}=\lim\limits_{t\to 0}\dfrac{t}{x-1}=\lim\limits_{t\to 0}\dfrac{1}{x'}=\dfrac{1}{1+2}=\dfrac{1}{3}.$$
$$ x_{10}=\lim\limits_{t\to 0}\dfrac{y-1}{x-1}=\lim\limits_{t\to 0}\dfrac{y'}{x'}=\dfrac{2+1}{1+2}=1.$$
Из последних выражений для $y$ и $x$ находим
$$ x'=\dfrac{y_1-ty_1'}{y_1^2},$$
$$ y'=\dfrac{(x_1+tx_1')y_1-tx_1y_1'}{y_1^2}.$$
Подставим выражения для производных в исходное уравнение:
$$ \dfrac{y_1-ty_1'}{y_1^2}=2t+1+\dfrac{x_1t}{y_1}+2+\dfrac{2t}{y_1},$$
$$ y_1-ty_1'=2ty_1^2+3ty_1^3+tx_1y_1+2ty_1,$$
$$ y_1'=-2y_1^2-x_1y_1-2y_1+\dfrac{1}{t}(y_1-3y_1^2).$$
Аналогично
$$\dfrac{(x_1+tx_1')y_1-tx_1y_1'}{y_1^2}=t+2+\dfrac{2x_1t}{y_1}+1+\dfrac{t}{y_1}, $$

$$(x_1+tx_1')y_1=-2x_1y_1^2t-x_1^2y_1t-2tx_1y_1+x_1y_1-3x_1y_1^2+ty_1^2+3y_1^2+2x_1y_1t+ty_1$$
$$x_1+tx_1'=x_1-2x_1y_1t-tx_1^2-3x_1y_1+ty_1+t+3y_1,$$
$$x_1'=-2x_1y_1-x_1^2+y_1+1+\dfrac{1}{t}(3y_1-3x_1y_1).$$
Таким образом, имеем систему
\begin{equation*}
\left\{

\end{aligned}
\right.
\end{equation*}

$$x_1=1+\dfrac{t}{y_2},~y_1=\dfrac{1}{3}+\dfrac{tx_2}{y_2}$$

Вычислим отдельно $\lim\limits_{t\to 0}x_1':$
$$\lim\limits_{t\to 0}x_1'=\lim\limits_{t\to 0}\dfrac{3y_1-3x_1y_1}{t}+\dfrac{1}{3}+1-\dfrac{2}{3}-1=-\dfrac{1}{3}+3\lim\limits_{t\to 0}\dfrac{y_1(1-x_1)}{t}$$
$$\lim\limits_{t\to 0}x_1'=-\dfrac{1}{3}-\lim\limits_{t\to 0}x_1'.$$
$$2\lim\limits_{t\to 0}x_1'=-\dfrac{1}{3}$$
$$\lim\limits_{t\to 0}x_1'=-\dfrac{1}{6}$$
Тогда $y_{20}=-6$.
$$x_{20}=\lim\limits_{t\to 0}\dfrac{y_1'}{x_1'}=-6\lim\limits_{t\to 0}y_1'=-6(-\dfrac{2}{3}-\dfrac{1}{3}-\dfrac{2}{9}+\lim\limits_{t\to 0}\dfrac{y_1(1-3y_1)}{t})=-6(-\dfrac{11}{9}-\lim\limits_{t\to 0}y_1')$$
Тогда $\lim\limits_{t\to 0}y_1'=-\dfrac{11}{9}-\lim\limits_{t\to 0}y_1'$, то есть $\lim\limits_{t\to 0}y_1'=-\dfrac{11}{18}$. Получаем, что $$x_{20}=6\cdot \dfrac{11}{18}=\dfrac{11}{3}.$$
Найдём производные:

$$y_2't=y_2+\dfrac{y_2^2}{3}+x_2y_2t+\dfrac{8y_2t}{3}+2x_2t^2+t^2+y_2+3x_2t,$$

Рассмотрим второе уравнение:

\underline{Пример 3.} 
\begin{center}
\begin{equation}
\left\{
\begin{aligned}
x'=y,~~x(0)=0, \\
y'=\sin(x+y),~~y(0)=\dfrac{\pi}{2} \\
\end{aligned}
\right.
\end{equation}
\end{center}

Решим данную систему методом разложения решения в цепную дробь и сравним с решением, полученным методом Рунге-Кутта.

Введём новые переменные по формулам
$$y_1=\dfrac{t}{x},~x_1=\dfrac{y-\dfrac{\pi}{2}}{x}, $$
или
$$x=\dfrac{t}{y_1},~y=\dfrac{\pi}{2}+\dfrac{tx_1}{y_1}. $$
Находим $y_{10}$ и $x_{10}$:
$$ y_{10}=\lim\limits_{t\to 0}\dfrac{t}{x}=\lim\limits_{t\to 0}\dfrac{1}{x'}=\dfrac{2}{\pi}.$$
$$ x_{10}=\lim\limits_{t\to 0}\dfrac{y-\dfrac{\pi}{2}}{x}=\lim\limits_{t\to 0}\dfrac{y'}{x'}=\lim\limits_{t\to 0}\dfrac{\sin(x+y)}{y}=\dfrac{2}{\pi}.$$
Из последних выражений для $y$ и $x$ находим
$$ x'=\dfrac{y_1-ty_1'}{y_1^2},$$
$$ y'=\dfrac{(x_1+tx_1')y_1-tx_1y_1'}{y_1^2}.$$
Подставим выражения для производных в исходное уравнение:
$$ \dfrac{y_1-ty_1'}{y_1^2}=\dfrac{tx_1}{y_1}+\dfrac{\pi}{2},$$
$$ y_1-ty_1'=tx_1y_1+\dfrac{\pi}{2}y_1^2,$$
$$ y_1'=\dfrac{y_1}{t}-x_1y_1-\dfrac{\pi y_1^2}{2t}.$$
Аналогично
$$\dfrac{(x_1+tx_1')y_1-tx_1y_1'}{y_1^2}=\sin(\dfrac{t}{y_1}+\dfrac{tx_1}{y_1}+\dfrac{\pi}{2}), $$
$$(x_1+tx_1')y_1-tx_1y_1'=ty_1^2 \sin(\dfrac{tx_1}{y_1}+\dfrac{\pi}{2}),$$
$$(x_1+tx_1')y_1=tx_1(\dfrac{y_1}{t}-x_1y_1-\dfrac{\pi y_1^2}{2t})+y_1^2 \sin(\dfrac{tx_1}{y_1}+\dfrac{\pi}{2})$$
$$(x_1+tx_1')y_1=x_1y_1-tx_1^2y_1-\dfrac{\pi}{2}x_1y_1^2+y_1^2 \sin(\dfrac{tx_1}{y_1}+\dfrac{\pi}{2})$$
Таким образом, имеем систему
\begin{equation*}
\left\{
\begin{aligned}
x_1'=-x_1^2-\dfrac{\pi x_1y_1^2}{2}+y_1^2 \sin(\dfrac{tx_1}{y_1}+\dfrac{\pi}{2}),~x_{10}=x_1(0)=\dfrac{2}{\pi},\\
y_1'=\dfrac{y_1}{t}-x_1y_1-\dfrac{\pi y_1^2}{2t},~y_{10}=y_1(0)=\dfrac{2}{\pi}.
\end{aligned}
\right.
\end{equation*}
Опять введём новые переменные
$$y_2=\dfrac{t}{x_1-x_{10}}=\dfrac{t}{x_1-\dfrac{2}{\pi}},~~x_2=\dfrac{y_1-\dfrac{2}{\pi}}{x_1-\dfrac{2}{\pi}},$$
или
$$x_1=\dfrac{2}{\pi}+\dfrac{t}{y_2},~y_1=\dfrac{2}{\pi}+\dfrac{tx_2}{y_2}$$
Находим $y_{20}$ и $x_{20}$:
$$y_{20}=\lim\limits_{t\to 0}\dfrac{t}{x_1-1}=\lim\limits_{t\to 0}\dfrac{1}{x_1'}$$
Вычислим отдельно $\lim\limits_{t\to 0}x_1':$
$$\lim\limits_{t\to 0}x_1'=\lim\limits_{t\to 0}\dfrac{-\dfrac{\pi}{2}x_1y_1+y_1\sin(\dfrac{tx_1}{y_1}+\dfrac{\pi}{2})}{t}-\dfrac{2^2}{\pi^2}=$$
$$=-\dfrac{2^2}{\pi^2}+\lim\limits_{t\to 0}\dfrac{y_1(-\dfrac{\pi}{2}x_1+1)}{t}=-\dfrac{2^2}{\pi^2}+\dfrac{2}{\pi}\lim\limits_{t\to 0}\dfrac{-\dfrac{\pi}{2}x_1'}{1}$$
$$\lim\limits_{t\to 0}x_1'=-\dfrac{4}{\pi^2}-\lim\limits_{t\to 0}x_1'.$$
$$2\lim\limits_{t\to 0}x_1'=-\dfrac{4}{\pi^2}$$
$$\lim\limits_{t\to 0}x_1'=-\dfrac{2}{\pi^2}$$
Тогда $y_{20}=-\dfrac{\pi^2}{2}$.
$$x_{20}=\lim\limits_{t\to 0}\dfrac{y_1'}{x_1'}=-\dfrac{\pi^2}{2}\lim\limits_{t\to 0}y_1'=-\dfrac{\pi^2}{2}(\dfrac{2}{\pi}\cdot\dfrac{2}{\pi}+\lim\limits_{t\to 0}\dfrac{y_1-\dfrac{\pi}{2}y_1^2}{t})=-\dfrac{\pi^2}{2}(-\dfrac{4}{\pi^2}-\lim\limits_{t\to 0}y_1')$$
Тогда $\lim\limits_{t\to 0}y_1'=-\dfrac{4}{\pi^2}-\lim\limits_{t\to 0}y_1'$, то есть $\lim\limits_{t\to 0}y_1'=-\dfrac{2}{\pi^2}$. Получаем, что $$x_{20}=1.$$

\end{bmatrix}
$$

0&1&\dfrac{\pi}{2}&\dfrac{2}{\pi}t+1&t^2-\dfrac{\pi}{2}t-\dfrac{\pi^2}{2}\\
1&0&0&t&-\dfrac{\pi^2}{2}-t\\
0&0&1&\dfrac{2}{\pi}&t-\pi
\end{array}

$$y_{прибл}=\dfrac{t^2-\dfrac{\pi}{2}t-\dfrac{\pi^2}{2}}{t-\pi},$$
$$x_{прибл}=\dfrac{-\dfrac{\pi^2}{2}t}{t-\pi}$$
